001package pkg_core;
002
003import java.util.ArrayList;
004public class ItemList
005{
006    private ArrayList<Item> items;
007
008    public ItemList(){
009        items = new ArrayList<Item>(0);
010    }
011
012    public Item getItem(String s){
013        Item res = null;
014        for(Item i : items){
015            if(i.getName().equals(s))
016                res = i;
017        }
018        return res;
019    }
020
021    public void addItem(Item i){
022        items.add(i);
023    }
024
025    public void removeItem(String s){
026        Item r = null;
027        for(Item i : items){
028            if(i.getName().equals(s)){
029                r = i;
030            }
031        }
032        if(r != null)
033            items.remove(r);
034    }
035    
036    public String desc(){
037        String res = "";
038        for(Item i : items){
039            res += "        |"+i.getLongDescription()+"\n";
040        }
041        if(res.equals(""))
042            return "+++pas d'objets+++";
043        return res;
044    }
045    
046    public boolean hasItem(String s){
047        boolean res = false;
048        for(Item i : items){
049            if(i.getName().equals(s)){
050                res = true;
051            }
052        }
053        return res;
054    }
055    
056} // ItemList
