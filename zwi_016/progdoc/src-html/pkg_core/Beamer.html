001package pkg_core;
002
003public class Beamer extends Item
004{
005    private Room chargedRoom;
006    private boolean charged = false;
007    public Beamer(){
008        super("beamer",30,"Un portal gun de salles");
009    }
010    
011    public Room getChargedRoom(){
012        return this.chargedRoom;
013    }
014    
015    @Override public String charge(){
016        if(this.charged == true)
017            return("+++problème+++beamer déja chargé+++");
018        chargedRoom = player.getRoom();
019        this.charged = true;
020        return("beamer chargé");
021    }
022
023    @Override public String use(){
024        if(chargedRoom == null)
025            return "+++impossible+++beamer non chargé+++";
026        player.setRoom(chargedRoom);
027        this.chargedRoom = null;
028        player.cutStack();
029        return("beamer utilisé");
030    }
031}
