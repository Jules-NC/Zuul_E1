001package pkg_commands; 
002
003import pkg_core.Player;
004
005public abstract class Command
006{
007    protected String secondWord;
008
009    /**
010     * Create a command object. First and second words must be supplied, but
011     * the second may be null.
012     * @param commandWord The CommandWord. UNKNOWN if the command word
013     *                  was not recognised.
014     * @param secondWord The second word of the command. May be null.
015     */
016    public Command()
017    {
018        this.secondWord = null;
019    }
020
021    /**
022     * @return The second word of this command. Returns null if there was no
023     * second word.
024     */
025    public String getSecondWord()
026    {
027        return secondWord;
028    }
029
030    public void setSecondWord(String secondWord)
031    {
032        this.secondWord = secondWord;
033    }
034
035    /**
036     * @return true if the command has a second word.
037     */
038    public boolean hasSecondWord()
039    {
040        return (secondWord != null);
041    }
042
043    public abstract void execute(Player player);
044}
